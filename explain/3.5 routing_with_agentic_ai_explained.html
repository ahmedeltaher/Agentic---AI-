💻 الكود العملي: Router مع AeroFlow و EcoSprint

ده الكود الفعلي اللي بيوضح إزاي Router بيشتغل مع محركات الاستعلام:

from llama_index.core.agent import ReActAgent
from llama_index.core.tools import QueryEngineTool
from llama_index.core import VectorStoreIndex, Document
from llama_index.llms.openai import OpenAI

# Setup LLM with verbose mode enabled
llm = OpenAI(model="gpt-4", temperature=0)

# AeroFlow product documents
aeroflow_docs = [
    Document(text="AeroFlow is available in coastal blue, sunset orange, and pearl white colors."),
    Document(text="AeroFlow features advanced aerodynamic design for maximum performance."),
    Document(text="AeroFlow pricing starts at $299 for basic model, $499 for premium."),
    Document(text="AeroFlow comes with 5-year warranty and free shipping.")
]

# EcoSprint product documents  
ecosprint_docs = [
    Document(text="EcoSprint comes in forest green, earth brown, and sky blue colors."),
    Document(text="EcoSprint is made from 100% recycled materials for sustainability."),
    Document(text="EcoSprint is priced at $199 standard, $349 for deluxe version."),
    Document(text="EcoSprint includes 3-year warranty and carbon-neutral shipping.")
]

# Create vector indexes
print("📚 Creating vector indexes...")
aeroflow_index = VectorStoreIndex.from_documents(aeroflow_docs)
ecosprint_index = VectorStoreIndex.from_documents(ecosprint_docs)

# Create query engines
aeroflow_engine = aeroflow_index.as_query_engine()
ecosprint_engine = ecosprint_index.as_query_engine()
print("⚙️ Query engines created successfully!")

# Create router tools with detailed descriptions
aeroflow_tool = QueryEngineTool.from_defaults(
    query_engine=aeroflow_engine,
    name="aeroflow_query_engine",
    description=(
        "Use this tool to answer questions about AeroFlow product. "
        "This tool contains information about AeroFlow colors, pricing, "
        "features, specifications, warranty, and shipping details. "
        "Use this when the query specifically mentions AeroFlow or when "
        "asking about high-performance aerodynamic products."
    )
)

ecosprint_tool = QueryEngineTool.from_defaults(
    query_engine=ecosprint_engine,
    name="ecosprint_query_engine",
    description=(
        "Use this tool to answer questions about EcoSprint product. "
        "This tool contains information about EcoSprint colors, pricing, "
        "eco-friendly features, materials, sustainability aspects, "
        "warranty, and shipping. Use this when the query mentions "
        "EcoSprint or asks about environmentally friendly products."
    )
)

# Create router agent with verbose mode
print("🤖 Creating router agent...")
router_agent = ReActAgent.from_tools(
    tools=[aeroflow_tool, ecosprint_tool],
    llm=llm,
    verbose=True  # Enable detailed explanation of routing decisions
)

print("🚀 Router setup complete! Ready for queries...\n")

# Example 1: Query about AeroFlow colors
print("=" * 60)
print("🎯 QUERY 1: AeroFlow Colors")
print("=" * 60)

query1 = "What colors are available for AeroFlow?"
print(f"Question: {query1}")
print("\n🧠 Router thinking process:")

# Execute query and observe routing
response1 = router_agent.query(query1)

print(f"\n📋 Final Response: {response1}")
print(f"✅ Expected: coastal blue, sunset orange, pearl white")

# Example 2: Query about EcoSprint colors
print("\n" + "=" * 60)
print("🌱 QUERY 2: EcoSprint Colors") 
print("=" * 60)

query2 = "What colors are available for EcoSprint?"
print(f"Question: {query2}")
print("\n🧠 Router thinking process:")

# Execute query and observe routing
response2 = router_agent.query(query2)

print(f"\n📋 Final Response: {response2}")
print(f"✅ Expected: forest green, earth brown, sky blue")

# Function to demonstrate routing decision analysis
def analyze_routing_decision(query: str, expected_engine: str):
    """
    Analyze and explain routing decisions
    """
    print(f"\n🔍 ROUTING ANALYSIS")
    print(f"Query: '{query}'")
    
    # Simulate analysis
    query_lower = query.lower()
    
    if "aeroflow" in query_lower:
        predicted_engine = "aeroflow_query_engine"
        reasoning = "Query contains 'AeroFlow' keyword"
        confidence = "High"
    elif "ecosprint" in query_lower:
        predicted_engine = "ecosprint_query_engine"
        reasoning = "Query contains 'EcoSprint' keyword"
        confidence = "High"
    elif "eco" in query_lower or "green" in query_lower:
        predicted_engine = "ecosprint_query_engine"
        reasoning = "Query relates to environmental features"
        confidence = "Medium"
    elif "performance" in query_lower or "aerodynamic" in query_lower:
        predicted_engine = "aeroflow_query_engine"
        reasoning = "Query relates to performance features"
        confidence = "Medium"
    else:
        predicted_engine = "unclear"
        reasoning = "No clear indicators"
        confidence = "Low"
    
    print(f"🎯 Predicted Engine: {predicted_engine}")
    print(f"💭 Reasoning: {reasoning}")
    print(f"📊 Confidence: {confidence}")
    
    if predicted_engine == expected_engine:
        print("✅ Prediction matches expected result!")
    else:
        print("❌ Prediction differs from expected result")
    
    return predicted_engine

# Advanced router with learning capabilities
class SmartRouter:
    def __init__(self, tools):
        self.tools = tools
        self.routing_history = []
        self.performance_metrics = {}
    
    def route_with_learning(self, query: str):
        """
        Route query and learn from decisions
        """
        # Analyze query
        analysis = self.analyze_query_intent(query)
        
        # Select tool based on analysis
        selected_tool = self.select_optimal_tool(analysis)
        
        # Record decision
        decision_record = {
            "query": query,
            "analysis": analysis,
            "selected_tool": selected_tool,
            "timestamp": datetime.now()
        }
        self.routing_history.append(decision_record)
        
        return selected_tool
    
    def analyze_query_intent(self, query: str):
        """
        Analyze query intent with multiple factors
        """
        analysis = {
            "keywords": [],
            "intent": "unknown",
            "confidence": 0.0,
            "context_hints": []
        }
        
        query_lower = query.lower()
        
        # Keyword analysis
        aeroflow_keywords = ["aeroflow", "aerodynamic", "performance", "speed"]
        ecosprint_keywords = ["ecosprint", "eco", "green", "sustainable", "environment"]
        
        analysis["keywords"] = [kw for kw in aeroflow_keywords + ecosprint_keywords 
                               if kw in query_lower]
        
        # Intent determination
        if any(kw in query_lower for kw in aeroflow_keywords):
            analysis["intent"] = "aeroflow"
            analysis["confidence"] = 0.9
        elif any(kw in query_lower for kw in ecosprint_keywords):
            analysis["intent"] = "ecosprint"
            analysis["confidence"] = 0.9
        elif "color" in query_lower:
            analysis["intent"] = "color_inquiry"
            analysis["confidence"] = 0.6
        elif "price" in query_lower or "cost" in query_lower:
            analysis["intent"] = "pricing_inquiry"
            analysis["confidence"] = 0.7
        
        return analysis
    
    def select_optimal_tool(self, analysis):
        """
        Select the best tool based on analysis
        """
        if analysis["intent"] == "aeroflow":
            return "aeroflow_query_engine"
        elif analysis["intent"] == "ecosprint":
            return "ecosprint_query_engine"
        elif analysis["confidence"] < 0.5:
            return "general_fallback"
        else:
            # Use historical performance to decide
            return self.select_based_on_history(analysis)
    
    def select_based_on_history(self, analysis):
        """
        Use routing history to make better decisions
        """
        # Simplified logic for demonstration
        if len(self.routing_history) == 0:
            return "aeroflow_query_engine"  # Default
        
        # Find similar past queries
        similar_queries = [
            record for record in self.routing_history
            if any(kw in record["query"].lower() 
                  for kw in analysis["keywords"])
        ]
        
        if similar_queries:
            # Use most common selection from similar queries
            tool_counts = {}
            for record in similar_queries:
                tool = record["selected_tool"]
                tool_counts[tool] = tool_counts.get(tool, 0) + 1
            
            return max(tool_counts, key=tool_counts.get)
        
        return "aeroflow_query_engine"  # Default fallback

💡 نصائح لتحسين أداء Router

  • 🎯 استخدم كلمات مفتاحية واضحة: حط الكلمات اللي المستخدمين محتمل يستخدموها
  • 📋 اكتب أوصاف شاملة: اذكر كل الموضوعات اللي المحرك يقدر يجاوب عليها
  • 🔄 جرب وحسن: اختبر Router بأسئلة مختلفة وحسن الأوصاف
  • 🗣️ فعل verbose mode: علشان تفهم إزاي Router بياخد قراراته
  • ⚡ خلي الأوصاف مختلفة: كل محرك يكون له هوية مميزة
المعادلة الذهبية: وصف دقيق + كلمات مفتاحية واضحة + اختبار مستمر = Router ذكي ودقيق!
