💻 مثال مبسط: نظام توجيه المطالبات

class ClaimsRoutingAgent:
    def __init__(self):
        self.automated_agent = AutomatedClaimsAgent()
        self.manual_system = ManualClaimsSystem()
    
    def route_claim(self, claim, policy):
        # تحليل تعقيد المطالبة
        complexity = self._analyze_complexity(claim, policy)
        
        if complexity == 'simple':
            return self.automated_agent.process(claim, policy)
        else:
            return self.manual_system.forward(claim, policy)
    
    def _analyze_complexity(self, claim, policy):
        # معايير التعقيد
        if claim.amount > policy.max_auto_amount:
            return 'complex'
        if claim.type in ['injury', 'legal']:
            return 'complex'
        return 'simple'

class AutomatedClaimsAgent:
    def process(self, claim, policy):
        # خطوات المعالجة الآلية
        fraud_check = self._check_fraud(claim)
        verification = self._verify_claim(claim)
        settlement = self._calculate_settlement(claim, policy)
        
        if fraud_check['risk'] > 0.7:
            return {'status': 'rejected', 'reason': 'fraud_detected'}
        
        return {
            'status': 'approved',
            'settlement_amount': settlement,
            'processing_time': '2_hours'
        }
